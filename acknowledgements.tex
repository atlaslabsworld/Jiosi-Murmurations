\begin{center}
    \textit{"Tho much is taken, much abides;\\
    \& tho we are not now that strength,\\
    which in old days moved Earth \& Heaven,\\
    that which we are, we are:\\
    one equal temper of heroic hearts,\\
    made weak by time \& fate,\\
    but strong in will,\\
    to strive, to seek, to find,\\
    \& not to yield."}
\end{center}
